Структура конфигурационного файла:
```
reports_config_<...>_.json
├── general # Объект с общими настройками конфигурации отчётов
│   ├── auto_update_refs # Флаг, указывающий обновятся ли все описанные справочники автоматически
│   └── dump_mart_to_file # Флаг, указывающий сохраняется ли витрина данных в файл
├── activated_reports # Массив строк, содержащий идентификаторы активных отчётов
├── all_reports # Массив строк, перечисляющий идентификаторы всех доступных отчётов
├── cols_configuration # Объект, дополнительно определяющий типы данных для столбцов, используемых в отчётах
│   └── <col_name>: <col_type:str> # Тип данных для столбца месяца отчёта (целое число)
├── refs_cleanup_config # Объект с настройками очистки справочных данных
│   ├── strip # Объект для настройки удаления крайних пробелов и непечатных символов
│   │   └── excludes # Массив строк, исключения для удаления пробелов
│   └── replaces # Массив объектов, описывающих замену символов
│       └── [n] # Объект для одной замены символов
│           ├── what # Строка, символ или последовательность для замены
│           ├── to # Строка, на которую выполняется замена
│           ├── excludes # Массив идентификаторов справочников, исключенных из замены
│           └── includes # Массив идентификаторов справочников, включенных для замены
├── sources # Объект, описывающий источники данных для отчётов
│   ├── categorical_not_groups_cols # Массив объектов, описывающих категориальные столбцы
│   │   └── [n] # Объект для одного категориального столбца
│   │       ├── name # Строка, название столбца
│   │       └── using_ref # Строка, идентификатор используемого справочника
│   ├── attributes # Объект с конфигурацией атрибутов групп
│   │   └── <source_name> # Объект с настройками для конкретного источника атрибутов
│   │       ├── columns # Массив строк, перечень столбцов таблицы-источника атрибутов
│   │       ├── primary_key_columns # Массив строк, столбцы первичного ключа
│   │       └── additional_columns # Объект с дополнительными столбцами
│   │           └── <column_name>: <value:sql> # Вычисляемое поле, строка с выражением 
│   └── amounts # Объект с конфигурацией данных о результатах по группам
│       └── <source_name> # Объект с настройками для конкретного источника результатов
│           ├── columns # Массив строк, перечень столбцов таблицы-источника результатов
│           ├── primary_key_columns # Массив строк, столбцы первичного ключа
│           └── amount_columns # Массив строк, столбцы с данными сумм
├── constructors # Объект с конфигурацией файлов-конструкторов для построения отчётов
│   └── <constructor_id> # Объект с настройками конструктора 
│       ├── filename # Строка, имя файла Excel с конструктором
│       ├── constructor_sheet_name # Строка, имя листа в файле Excel
│       └── header # Целое число, номер строки заголовка в файле
├── reports # Объект с конфигурацией отдельных отчётов
│   └── <report_id> # Объект с настройками для конкретного отчёта
│       ├── report_name # Строка, имя отчёта
│       ├── group_attributes_source # Строка, идентификатор источника атрибутов групп
│       ├── group_amounts_source # Строка, идентификатор источника сумм групп
│       ├── initial_columns # Массив строк, начальные столбцы для отчёта
│       └── mart_structure # Объект, описывающий структуру витрины данных
│           ├── table_name # Строка, имя таблицы витрины
│           ├── columns # Массив строк, столбцы витрины
│           ├── using_constructor # Строка, идентификатор используемого конструктора
│           └── using_refs # Массив строк, идентификаторы используемых справочников
└── refs # Объект с конфигурацией справочников
    └── <ref_id> # Объект с настройками конкретного справочника
        ├── filename # Строка, имя файла Excel со справочником
        ├── header # Целое число, номер строки заголовка
        ├── cd_col # Строка, название столбца с кодом
        ├── name_col # Строка, название столбца с именем
        └── where_condition # Строка, условие фильтрации данных (опционально)
```
